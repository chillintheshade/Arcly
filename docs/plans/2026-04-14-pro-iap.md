# PieMenu Pro 内购实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现免费 + 一次性内购解锁 Pro 的盈利模式，免费版限制 4 个应用槽位，Pro 版解锁全部 12 个槽位、自定义快捷键、音乐控制、外观设置等高级功能。

**Architecture:** 新建 `StoreManager` 单例处理 StoreKit 2 购买逻辑，在 `AppState` 中暴露 `isPro` 状态，各 View 根据此状态显示/隐藏功能或展示升级提示。内购状态由 StoreKit 2 的 `Transaction.currentEntitlements` 管理，无需自建服务器。

**Tech Stack:** StoreKit 2, SwiftUI, Combine

### Task 1: StoreManager — StoreKit 2 购买核心

**Files:**
- Create: `Sources/PieMenu/StoreManager.swift`

- [ ] **Step 1: 创建 StoreManager**

```swift
import StoreKit

@MainActor
final class StoreManager: ObservableObject {
    static let shared = StoreManager()

    static let proProductID = "com.piemenu.app.pro"

    @Published private(set) var isPro: Bool = false
    @Published private(set) var proProduct: Product?
    @Published private(set) var purchaseError: String?

    private var transactionListener: Task<Void, Never>?

    private init() {
        transactionListener = listenForTransactions()
        Task { await loadProducts() }
        Task { await updatePurchaseStatus() }
    }

    deinit {
        transactionListener?.cancel()
    }

    // MARK: - 加载商品

    func loadProducts() async {
        do {
            let products = try await Product.products(for: [Self.proProductID])
            proProduct = products.first
        } catch {
            NSLog("❌ 加载商品失败: %@", error.localizedDescription)
        }
    }

    // MARK: - 购买

    func purchasePro() async {
        guard let product = proProduct else {
            purchaseError = "商品信息加载中，请稍后再试"
            return
        }
        do {
            let result = try await product.purchase()
            switch result {
            case .success(let verification):
                let transaction = try checkVerified(verification)
                await transaction.finish()
                isPro = true
                purchaseError = nil
            case .userCancelled:
                break
            case .pending:
                purchaseError = "购买等待审批中"
            @unknown default:
                break
            }
        } catch {
            purchaseError = "购买失败: \(error.localizedDescription)"
        }
    }

    // MARK: - 恢复购买

    func restorePurchases() async {
        try? await AppStore.sync()
        await updatePurchaseStatus()
    }

    // MARK: - 检查购买状态

    func updatePurchaseStatus() async {
        for await result in Transaction.currentEntitlements {
            if let transaction = try? checkVerified(result),
               transaction.productID == Self.proProductID {
                isPro = true
                return
            }
        }
        isPro = false
    }

    // MARK: - 监听交易更新

    private func listenForTransactions() -> Task<Void, Never> {
        Task.detached {
            for await result in Transaction.updates {
                if let transaction = try? self.checkVerified(result) {
                    await transaction.finish()
                    await self.updatePurchaseStatus()
                }
            }
        }
    }

    private func checkVerified<T>(_ result: VerificationResult<T>) throws -> T {
        switch result {
        case .unverified(_, let error):
            throw error
        case .verified(let item):
            return item
        }
    }
}
```

### Task 4: PaywallView — 升级弹窗

**Files:**

- [ ] **Step 1: 创建 PaywallView**

```swift
import SwiftUI
import StoreKit

struct PaywallView: View {
    @ObservedObject private var store = StoreManager.shared
    @Binding var isPresented: Bool
    var highlightFeature: String?

    var body: some View {
        VStack(spacing: 20) {
            // 标题
            VStack(spacing: 8) {
                Image(systemName: "star.circle.fill")
                    .font(.system(size: 40))
                    .foregroundStyle(.yellow, .orange)

                Text("升级到 PieMenu Pro")
                    .font(.system(size: 18, weight: .bold))

                Text("解锁全部功能")
                    .font(.system(size: 13))
                    .foregroundStyle(.secondary)
            }

            // 功能列表
            VStack(alignment: .leading, spacing: 10) {
                featureRow("最多 12 个应用快捷方式", highlight: highlightFeature == "apps")
                featureRow("自定义快捷键", highlight: highlightFeature == "hotkey")
                featureRow("音乐播放控制", highlight: highlightFeature == "music")
                featureRow("外观自定义（半径、图标大小、主题）", highlight: highlightFeature == "appearance")
                featureRow("文件夹快捷方式", highlight: highlightFeature == "folder")
            }
            .padding(.horizontal, 12)

            // 购买按钮
            if let product = store.proProduct {
                Button(action: {
                    Task { await store.purchasePro() }
                }) {
                    Text("一次性购买 \(product.displayPrice)")
                        .font(.system(size: 14, weight: .semibold))
                        .frame(maxWidth: .infinity)
                        .padding(.vertical, 10)
                }
                .buttonStyle(.borderedProminent)
                .controlSize(.large)
            } else {
                ProgressView()
                    .controlSize(.small)
            }

            if let error = store.purchaseError {
                Text(error)
                    .font(.system(size: 11))
                    .foregroundColor(.red)
            }

            // 恢复购买
            Button("恢复购买") {
                Task { await store.restorePurchases() }
            }
            .font(.system(size: 12))
            .foregroundStyle(.secondary)
            .buttonStyle(.plain)
        }
        .padding(24)
        .frame(width: 320)
        .onChange(of: store.isPro) { newValue in
            if newValue { isPresented = false }
        }
    }

    private func featureRow(_ text: String, highlight: Bool = false) -> some View {
        HStack(spacing: 8) {
            Image(systemName: "checkmark.circle.fill")
                .font(.system(size: 14))
                .foregroundColor(.green)
            Text(text)
                .font(.system(size: 12))
                .foregroundColor(highlight ? .primary : .secondary)
                .fontWeight(highlight ? .medium : .regular)
        }
    }
}
```
